139 /**
140  * LoadManager は Image オブジェクトの読み込みを管理するクラスです。.
141  * @version 2.0.0 alpha
142  * @author Yasunobu Ikeda ( http://clockmaker.jp )
143  * @constructor
144  */
145 function LoadManager() {
146 }
147 
148 /**
149  *  LogLevel: すべてのログを出力します。
150  *  @type Number
151  *  @constant
152  */
153 LoadManager.LOG_VERBOSE = 0;
154 /**
155  * LogLevel: ログは一切出力しません。
156  * @type Number
157  * @constant
158  */
159 LoadManager.LOG_SILENT = 10;
160 /**
161  * LogLevel: エラーのときだけログを出力します。
162  * @type Number
163  * @constant
164  */
165 LoadManager.LOG_ERRORS = 4;
166 
167 /**
168  * スコープを移譲した関数を作成します。
169  * @param {Function} func 実行したい関数
170  * @param {Object} thisObj 移譲したいスコープ
171  * @return {Function} 移譲済みの関数
172  * @private
173  */
174 LoadManager._delegate = function (func, thisObj) {
175 	var del = function () {
176 		return func.apply(thisObj, arguments);
177 	};
178 	//情報は関数のプロパティとして定義する
179 	del.func = func;
180 	del.thisObj = thisObj;
181 	return del;
182 };
183 
184 LoadManager.prototype = {
185 
186 	/** URLリスト */
187 	_queueArr:[],
188 	_registerArr:[],
189 	_successArr:[],
190 	_errorArr:[],
191 	_isStarted:false,
192 	_isRunning:false,
193 	_isFinished:false,
194 	_currentQueues:[],
195 	_queueCount:0,
196 	/**
197 	 * 同時接続数です。デフォルトは6です。
198 	 * @type Number
199 	 * @default 6
200 	 */
201 	numConnections:6,
202 	/**
203 	 * プログレスイベント発生時に呼ばれるコールバックです。
204 	 * @params {lmlib.ProgressEvent} event ProgressEvent オブジェクトが第一引数で渡されます。
205 	 * @function
206 	 */
207 	onProgress:null,
208 	/**
209 	 * 処理完了時に呼ばれるコールバックです。
210 	 * @params {lmlib.CompleteEvent} event CompleteEvent オブジェクトが第一引数で渡されます。
211 	 * @function
212 	 */
213 	onComplete:null,
214 	/**
215 	 * エラー発生時に呼ばれるコールバックです。
216 	 * @params {lmlib.ErrorEvent} event  ErrorEvent オブジェクトが第一引数で渡されます。
217 	 * @function
218 	 */
219 	onError:null,
220 	/**
221 	 * ログのレベルを取得または設定します。デフォルトは LOG_SILENT (出力なし) です。
222 	 * @type Number
223 	 */
224 	logLevel:LoadManager.LOG_SILENT,
225 
226 	/**
227 	 * 処理が走っているかどうかを取得します。
228 	 * @return {Boolean}
229 	 */
230 	getIsRunning:function () {
231 		return this._isRunning;
232 	},
233 	/**
234 	 * 処理が完了しているかどうかを取得します。
235 	 * @return {Boolean}
236 	 */
237 	getIsFinished:function () {
238 		return this._isFinished;
239 	},
240 	/**
241 	 * 読み込みに成功したアイテムを配列として取得します。
242 	 * @return {Array} Array of LoadingData
243 	 */
244 	getSuccessItems:function () {
245 		return this._successArr;
246 	},
247 	/**
248 	 * 読み込みに失敗したアイテムを配列として取得します。
249 	 * @return {Array} Array of LoadingData
250 	 */
251 	getFailedItems:function () {
252 		return this._errorArr;
253 	},
254 	/**
255 	 * 処理の進行度を0～1の値で取得します。 (either IO Error).
256 	 * @return {Number}
257 	 */
258 	getPercent:function () {
259 		var percent = (this._successArr.length + this._errorArr.length) / this._registerArr.length;
260 		percent = Math.min(1, Math.max(0, percent));
261 		return percent;
262 	},
263 
264 	/**
265 	 * 読み込みたいアセットを登録します。
266 	 * @params {String} url	URLです。
267 	 * @params {Object} extra	任意の Object データです。
268 	 * @return {lmlib.LoadingData} 読み込みアイテムの情報です。
269 	 */
270 	add:function (url, extra) {
271 		if (this._isStarted) {
272 			this._log("既にLoadManagerインスタンスがstartしているため、add()することができません。", LoadManager.LOG_ERRORS);
273 		}
274 
275 		var item = new lmlib.LoadingData(url, extra);
276 		this._registerArr.push(item);
277 		return item;
278 	},
279 
280 	/**
281 	 * URLをキーとして読み込み済みアセットを取得します。
282 	 * なるべく読み込み完了後(onCompete発生後)に使ってください。
283 	 * @param {String} url	URLです。このURLと一致するアセットを返します。
284 	 * @return {lmlib.LoadingData}	読み込み情報を保持したデータです。
285 	 */
286 	get:function (url) {
287 		for (var i = 0; i < this._registerArr.length; i++) {
288 			if (this._registerArr[i].url == url) {
289 				return this._registerArr[i];
290 			}
291 		}
292 		return null;
293 	},
294 
295 	/**
296 	 * 読み込み処理を開始します。
297 	 * @param {Number} withConnections (default = null) — [optional] 同時読み込み数です。
298 	 */
299 	start:function (withConnections) {
300 		if (this._isStarted) {
301 			throw "既にLoadManagerインスタンスがstartしているため、start()することができません。";
302 		}
303 
304 		if (typeof(withConnections) == "number") {
305 			this.numConnections = withConnections;
306 		}
307 
308 		this._queueArr = this._registerArr.concat();
309 		this._execute(this._queueArr[0]);
310 		this._isStarted = true;
311 	},
312 
313 	_loadNextImage:function () {
314 		while (this.numConnections > this._queueCount) {
315 			if (this._queueArr.length == 0)
316 				break;
317 
318 			var nextQueue = this._queueArr.shift();
319 
320 			// Retrieve next filename in queue
321 			this._currentQueues.push(nextQueue);
322 
323 			this._execute(nextQueue);
324 
325 			this._queueCount++;
326 		}
327 
328 		// Busy loading
329 		this._isRunning = true;
330 	},
331 
332 	_execute:function (loadingData) {
333 		if (loadingData.image == null)
334 			loadingData.image = new Image();
335 
336 		var self = this;
337 
338 		// Already loaded
339 		loadingData.image.onload = function () {
340 			self._log("[LM - Success] " + loadingData.url, LoadManager.LOG_VERBOSE);
341 
342 			self._successArr.push(loadingData);
343 
344 			if (typeof(loadingData.callbackFunc) == "function") {
345 				loadingData.callbackFunc(loadingData.image);
346 			}
347 
348 			// Remove from queue
349 			self._queueCount--;
350 
351 			// dispatch Event
352 			if (typeof(self.onProgress) == "function") {
353 				var event = new lmlib.ProgressEvent();
354 				event.target = self;
355 				event.itemsLoaded = self._successArr.length;
356 				event.itemTotal = self._registerArr.length;
357 				event.percent = self.getPercent();
358 				event.data = loadingData;
359 				self.onProgress(event);
360 			}
361 
362 			// Continue loading
363 			if (!self._checkFinished()) self._delayFunc();
364 		};
365 
366 		loadingData.image.onerror = function () {
367 			self._log("[LM - Error] " + loadingData.url, LoadManager.LOG_VERBOSE);
368 
369 			// Remove from queue
370 			self._errorArr.push(loadingData);
371 
372 			// Remove from queue
373 			self._queueCount--;
374 
375 			// dispatch Event
376 			if (typeof(self.onError) == "function") {
377 				var event = new lmlib.ErrorEvent();
378 				event.target = this;
379 				event.data = loadingData;
380 				self.onError(event);
381 			}
382 
383 			// Continue loading
384 			if (!self._checkFinished()) self._delayFunc();
385 		};
386 
387 		loadingData.image.src = loadingData.url;
388 	},
389 
390 	_checkFinished:function () {
391 		// Queue finished?
392 		if (this._successArr.length + this._errorArr.length == this._registerArr.length) {
393 			// Loading finished
394 			this._isRunning = false;
395 			this._isFinished = true;
396 
397 			// dispatch Event
398 			if (typeof(this.onComplete) == "function") {
399 				var event = new lmlib.CompleteEvent();
400 				event.target = this;
401 				this.onComplete(event);
402 			}
403 
404 			return true;
405 		}
406 		return false;
407 	},
408 
409 	/** delay loading for stack over flow error of IE6 */
410 	_delayFunc:function () {
411 		setTimeout(LoadManager._delegate(this._loadNextImage, this), 16);
412 	},
413 
414 	_log:function (object, level) {
415 		if (this.logLevel <= level)
416 			console.log(object);
417 	}
418 };
419 window.LoadManager = LoadManager;
